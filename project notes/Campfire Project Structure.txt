│   │       └── ...
│   │   ├── services/             # FastAPI BL for auth
│   │       └── ...
│   │   ├── constants.py          # Pydantic Constants for auth
│   │   ├── dependencies.py       # Pydantic Dependencies for auth
│   │   └── validators.py         # Pydantic Validators for auth
│   │
│   ├── core/                     # Feature: Core
│   ├── event/                    # Feature: Event
│   ├── messaging/                # Feature: Messaging
│   ├── users/                    # Feature: Users
│   │
│   # ... other feature folders as needed ...
│   │
│   ├── core/                     # SHARED: Global configuration and security utilities
│   │   ├── config.py             # Application settings

│   │
│   └── main.py                   # SHARED: Main FastAPI application instance,
│                                 #         includes routers from all feature.api submodules
│
├── migrations/                   # Alembic migrations (manages all models that use app.db.base.Base)
